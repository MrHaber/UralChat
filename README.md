Неофициальный репозиторий плагина UralChat.

конфиг config.yml:
```yaml
# плагин был почти полностью пересан под нас https://github.com/lokha/UralChat
#
# Описание
# delays - разрешенный период написания команд
#          алиасы указывать не нужно! Если указали /gm, то не нужно прописывать /gamemode, плагин умный
# права:
# - UralChat.ignore позволяет писать команды без задержек
# - UralChat.admin - право перезагрузки конфига - /uralchat

delays:
  '/rtp': 20 # в секундах
 
DELAY_MESSAGE: '&4Команда будет доступна через <delay> секунд.'
```
